public void ToScene(SceneTag sTag)
{
    switch (sTag)
    {
    case SceneTag.FreeRun:
        GenGameModSelectEvent(SceneModTag.SceneModTag_1);
        break;
    case SceneTag.FreeRun2:
        GenGameModSelectEvent(SceneModTag.SceneModTag_2);
        break;
    case SceneTag.CNR:
        curSceneTag = sTag;
        StartCoroutine(StartSingleScene());
        return;
    }
    
    // V11 CHANGE: Wrap LoadLevel in coroutine to see if freeze happens after LoadLevel returns
    StartCoroutine(LoadLevelWithLogging(GetSceneNameByTag(sTag)));
    curSceneTag = sTag;
}

private IEnumerator LoadLevelWithLogging(string sceneName)
{
    UnityEngine.Debug.Log("[Jacqueb] V11-LoadLevelWithLogging START for scene: " + sceneName);
    
    Application.LoadLevel(sceneName);
    
    UnityEngine.Debug.Log("[Jacqueb] V11-LoadLevel() returned successfully");
    
    // Yield to let the new scene's Awake() run
    yield return null;
    
    UnityEngine.Debug.Log("[Jacqueb] V11-AFTER first yield null");
    
    // Yield one more time
    yield return null;
    
    UnityEngine.Debug.Log("[Jacqueb] V11-AFTER second yield null");
}
